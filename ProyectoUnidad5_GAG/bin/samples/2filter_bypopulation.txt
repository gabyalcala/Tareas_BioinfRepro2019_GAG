# Este script crea los archivos vcf de las poblaciones Taviche, Coste単o, Tusta y Chile de agua

## Correr vcftools en docker
## Crear una variable para correr docker estando en el directorio data

vcftools="docker run --rm -v /Users/gabyalcala/Desktop/Tareas_BioinfRepro2019_GAG/ProyectoUnidad5_GAG/data:/data biocontainers/vcftools:0.1.15 vcftools"

### make taviche population file 

$vcftools --vcf mexcollection.vcf --remove ./samples_names/only_taviche.txt --max-missing 0.9 --maf 0.05 --recode --recode-INFO-all --out taviche.vcf


### make coste単o population file

$vcftools --vcf mexcollection.vcf --remove ./samples_names/only_coste単o.txt --max-missing 0.09 --maf 0.05 --recode --recode-INFO-all --out coste単o.vcf


### make tusta population file
$vcftools --vcf mexcollection.vcf --remove ./samples_names/only_tusta.txt --max-missing 0.09 --maf 0.05 --recode --recode-INFO-all --out tusta.vcf

### make chile de agua population file 
$vcftools --vcf mexcollection.vcf --remove ./samples_names/only_chiledeagua.txt --max-missing 0.09 --maf 0.05 --recode --recode-INFO-all --out chiledeagua.vcf

### move .recode.vcf files to directory vcf_files_by_population

mv *.vcf.recode.vcf ./vcf_files_by_population/
